// ===== ОБНОВЛЁННАЯ ФУНКЦИЯ ДЛЯ РИСОВАНИЯ ОБРАЗЦА ===== function drawSampleChar(char) { const size = canvas.width; ctx.save(); // Очищаем область для образца (не стираем сетку) // Образец рисуется поверх сетки // Определяем размер шрифта в зависимости от количества символов let fontSize = size * 0.65; // базовый размер для 1 символа // Если символов больше 1, уменьшаем размер if (char.length >= 2) { fontSize = size * 0.45; } if (char.length >= 3) { fontSize = size * 0.32; } if (char.length >= 4) { fontSize = size * 0.25; } // Для слов из 2+ символов используем перенос по вертикали ctx.fillStyle = 'rgba(200, 180, 160, 0.25)'; ctx.textAlign = 'center'; ctx.textBaseline = 'middle'; ctx.font = `${fontSize}px "Noto Sans SC", "PingFang SC", "Microsoft YaHei", sans-serif`; if (char.length === 1) { // Один символ — по центру ctx.fillText(char, size/2, size/2 + size * 0.03); } else if (char.length === 2) { // Два символа — горизонтально ctx.fillText(char, size/2, size/2 + size * 0.03); } else { // Три и более — вертикально (столбиком) const chars = char.split(''); const lineHeight = fontSize * 1.1; const totalHeight = chars.length * lineHeight; const startY = (size - totalHeight) / 2 + fontSize * 0.3; for (let i = 0; i < chars.length; i++) { const y = startY + i * lineHeight; ctx.fillText(chars[i], size/2, y); } } ctx.restore(); }
